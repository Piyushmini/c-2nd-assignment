{
    class student
    {
        public static bool Is_palidrome(string s)
        {
            int st = 0;
            int en = s.Length-1;  //two pointer approach
            while (st < en)
            {
                if (s[st] != s[en])
                {
                    return false;
                }
                st++;
                en--;
            }
            return true;
        }
        static void Main(string[] args)
